页面方法：

 handleChange(info) {
      if (info.file.status === 'uploading') {
        this.attachment.uploading = true;
        return null;
      }
      if (info.file.status === 'error') {
        const {message} = info.file.response;
        this.$message.error(message);
        this.attachment.uploading = false;
        return null;
      }
      if (info.file.status === 'done') {
        const {id, path} = info.file.response;
        this.attachment.url = path;
        this.attachment.uploading = false;
        return id;
      }
      return null;
}

handleBeforeUpload(file) {
    const isJpgOrPng = file.type === 'image/jpeg' || file.type === 'image/png' || file.type === 'image/gif';
    if (!isJpgOrPng) {
        this.$message.error('只能上传图片！');
    }
    const isLt2M = file.size / 1024 / 1024 < 10;
    if (!isLt2M) {
        this.$message.error('图片大小必须小于 10MB！');
    }
    return isJpgOrPng && isLt2M;
},

后台接口：

@PostMapping("/uploadAttachment")
@RequiresPermission(value = "borrow.channelInstitutionVisiting.uploadAttachment", description = "我的客户-拜访记录-上传附件")
public ResponseEntity<?> upload(@RequestParam("file") MultipartFile file) throws IOException {
    UploadFile uploadFile = WebUtils.getUploadFile(file);
    UploadFileValidator uploadFileValidator = new UploadFileValidator(new String[]{
            "jpg", "png", "gif", "jpeg"
    }, 10 * 1024 * 1024);
    uploadFileValidator.validate(uploadFile);

    String path = fileStorage.store(uploadFile, "channelInstitutionVisiting");
    return ResponseEntity
            .ok(ImmutableMap.<String, Object>builder()
                    .put("path", path)
                    .build());
}

后台接口工具类：

public class UploadFileValidator {

    private String[] allowedExtensionNames;
    private long maxFileSize;

    public UploadFileValidator(String[] allowedExtensionNames, long maxFileSize) {
        this.allowedExtensionNames = allowedExtensionNames;
        this.maxFileSize = maxFileSize;
    }

    public void validate(UploadFile uploadFile) {
        if (!ArrayUtils.contains(allowedExtensionNames, uploadFile.getExtensionName().toLowerCase())) {
            throw new IllegalArgumentException("不支持的文件类型");
        }

        if (uploadFile.getSize() > maxFileSize) {
            throw new IllegalArgumentException("文件大小不能超过：" + FileUtils.byteCountToDisplaySize(maxFileSize));
        }
    }
}


public class WebUtils {

    public static String getClientIpAddress(HttpServletRequest request) {
        String xForwardedForHeader = request.getHeader(HttpHeaders.X_FORWARDED_FOR);
        if (xForwardedForHeader == null) {
            return request.getRemoteAddr();
        }
        return xForwardedForHeader.split(",")[0].trim();
    }

    public static UploadFile getUploadFile(MultipartFile file) throws IOException {
        UploadFile uploadFile = new UploadFile();
        uploadFile.setOriginalFilename(file.getOriginalFilename());
        uploadFile.setSize(file.getSize());
        uploadFile.setInputStream(file.getInputStream());
        uploadFile.setExtensionName(FilenameUtils.getExtension(uploadFile.getOriginalFilename()).toLowerCase());
        return uploadFile;
    }


    public static HttpServletRequest getHttpServletRequest() {
        return ((ServletRequestAttributes) RequestContextHolder.getRequestAttributes()).getRequest();
    }
}


public String store(UploadFile file, String directory) throws IOException {
    String filename = UUID.randomUUID().toString();
    String extension = FilenameUtils.getExtension(file.getOriginalFilename());
    File destDirectory = new File(diskFileStorageProperties.getBaseDirectory(), directory);
    String filenameWithExtension = filename + "." + extension;
    File destFile = new File(destDirectory, filenameWithExtension);
    if (!destDirectory.exists() && !destDirectory.mkdirs()) {
        throw new IOException("文件目录创建失败");
    }
    FileOutputStream os = new FileOutputStream(destFile);
    IOUtils.copy(file.getInputStream(), os);
    os.close();
    return diskFileStorageProperties.getBaseUrl() + "/" + directory + "/" + filenameWithExtension;
}

    }
}
