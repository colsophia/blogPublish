# 一：实现登录功能

# 1、两次MD5加密

  • 用户端：PASS=MD5(明文+固定Salt)

    • 用户端MD5加密是为了防止用户密码在网络中明文传输

  • 服务端：PASS=MD5(用户输入+随机Salt)

    • 服务端MD5加密是为了提交密码安全性，双重保险

# 2、创建表结构

create table t_user
(
    id              bigint           not null comment '用户id，手机号码' primary key,
    nickname        varchar(255)  not null,
    password        varchar(32)   not null comment 'MD5(MD5(明文+固定salt)+salt)',
    salt            varchar(10)   not null,
    register_date   datetime      not null comment '注册时间',
    last_login_date datetime      not null comment '最后一次登录时间',
    login_count     int default 0 null comment '登录次数',
    head            varchar(128)  null comment '头像'
)
    comment '用户表';

# 4、新建MD5工具类

public class MD5Util {

    public static String md5(String src) {
        return DigestUtils.md5Hex(src);
    }


    private static final String SALT = "NJGCc4T65n";

    /**
     * 客户端加密
     */
    public static String inputPassToFormPass(String inputPass) {
        String str = SALT.charAt(0) + SALT.charAt(1) + SALT.charAt(3) + inputPass + SALT.charAt(2) + SALT.charAt(4) + SALT.charAt(6);
        return md5(str);
    }


    /**
     * 服务端加密
     */
    public static String formPassToDBPass(String formPass, String salt) {
        String str = salt.charAt(0) + salt.charAt(1) + salt.charAt(3) + formPass + salt.charAt(2) + salt.charAt(4) + salt.charAt(6);
        return md5(str);
    }

    /**
     * 2次加密，存到数据库前进行MD5加密
     */
    public static String inputPassToFormPass(String inputPass, String salt) {
        return formPassToDBPass(inputPassToFormPass(inputPass), salt);
    }
}

# 5、登录校验逻辑：

@Override
public RespBean doLogin(String mobile, String password) {
    User user = userMapper.selectById(mobile);  // 采用BaseMapper 自带的根据id查找对象

    if (user == null) {
        throw new GlobalControllerException(RespBeanEnum.MOBILE_NOT_EXIST);
    }

    if (!MD5Util.formPassToDBPass(password, user.getSalt()).equals(user.getPassword())) {
        throw new GlobalControllerException(RespBeanEnum.LOGIN_ERROR);
    }

    return RespBean.success();
}
